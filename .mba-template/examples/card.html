Usage

<!-- Basic bordered card -->
<article class="card card-bordered">
  <div class="card-body">
    <h3 class="card-title">Card Title</h3>
    <p class="card-description">Card description text.</p>
  </div>
</article>

<!-- Card with header and footer -->
<article class="card card-bordered">
  <header class="card-header">
    <h3 class="card-header-title">Title</h3>
    <span class="card-badge card-badge-success">Active</span>
  </header>
  <div class="card-body">Content</div>
  <footer class="card-footer">
    <button class="btn btn-primary btn-sm">Action</button>
  </footer>
</article>

<!-- Stat card -->
<article class="card card-bordered stat-card">
  <p class="stat-label">Total Revenue</p>
  <p class="stat-value">$127,450</p>
  <p class="stat-change stat-change-positive">+12.4%</p>
</article>
